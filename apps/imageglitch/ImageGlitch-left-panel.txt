// Import Plugins
ai = {import:ai-text-plugin}
textToImage = {import:text-to-image-plugin}
remember = {import:remember-plugin}

// Aliases to ensure they exist in global scope under these names
pluginAi = ai
pluginTextToImage = textToImage

init
  [storeInstructionsToMemory]

storeInstructionsToMemory
  [remember.set("aiScribeInstruction", `[aiScribeInstruction]`)]
  [remember.set("aiChaosInstruction", `[aiChaosInstruction]`)]
  [remember.set("aiTransfigureInstruction", `[aiTransfigureInstruction]`)]

// --- INSTRUCTION INJECTION ---
// We inject the raw instructions into the HTML head as data scripts.
// This ensures they are available immediately without async database calls.
instructionInjection
  <script id="data-scribe" type="application/text">[aiScribeInstruction.selectAll.join('\n')]</script>
  <script id="data-chaos" type="application/text">[aiChaosInstruction.selectAll.join('\n')]</script>
  <script id="data-transfigure" type="application/text">[aiTransfigureInstruction.selectAll.join('\n')]</script>

// --- HTML BUILDER ---
buildImageGenerationHtml(data) =>
  var html = {import:apps/imageglitch/html/index.html}
  
  // Inject instructions into HEAD
  var htmlWithInstructions = html.replace("</head>", [instructionInjection] + "</head>")
  
  return htmlWithInstructions

doggo
  imgSeed = ""
  onFinish(result) =>
    let iframe = result.iframe;
    let canvas = result.canvas;
    iframe.replaceWith(canvas);

numImagesToGen = 0
masterCreativity = 4

// --- AI INSTRUCTIONS ---
aiScribeInstruction
  You are the "Holistic Prompt Architect".
  Your core goal is to take a simple idea and make it better by holistically filling in descriptive gaps.
  PROCESS:
  1. RECEIVE: Accept the user's base prompt.
  2. ANALYZE: Deconstruct the prompt.
  4. IDENTIFY GAPS: Determine which categories are lacking.
  5. HOLISTIC SELECTION: Select suitable terms from: [artisticStyles.selectOne], [composition.selectOne], [lighting.selectOne], [colorPalettes.selectOne], [mood.selectOne], [technicalDetails.selectOne].
  6. ENHANCE: Inject quality keywords: [aiCoreQuality.selectUnique(4).join(', ')], [aiFlavorEnhancers.selectUnique(3).join(', ')].
  7. SYNTHESIZE: Combine everything into a cohesive prompt.
  8. RETURN: Output ONLY the refined prompt string.

  IMPORTANT RULES:
  - Make the prompt 100% affirmative.
  - Return ONLY the prompt string.

aiChaosInstruction
  You are the "Mad Prompt Scientist".
  Your core goal is to spark new ideas by introducing random elements.
  PROCESS:
  1. RECEIVE: Accept user prompt.
  2. FILL GAPS: Fill lacking categories with RANDOM keywords: [artisticStyles.selectOne], [composition.selectOne], [lighting.selectOne], [colorPalettes.selectOne], [mood.selectOne].
  3. CHAOS TWIST: Always reroll at least one category.
  4. RETURN: Output ONLY the mutated prompt string.

aiTransfigureInstruction
  You are the "Prompt Modification Specialist".
  Your core goal is to surgically alter a prompt to match instructions.
  PROCESS:
  1. RECEIVE: Input A (Prompt), Input B (Instruction).
  2. MODIFY: Execute instructions precisely.
  3. RETURN: Output ONLY the modified prompt string.
